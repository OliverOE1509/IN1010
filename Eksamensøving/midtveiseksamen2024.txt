import java.util.Scanner;

class MinstAv5 {
    public static void main (String[] args) {
        Scanner tastatur = new Scanner(System.in);
        int min = 0;
        for (int i = 1; i <= 5; i++) {
            int v = tastatur.nextInt();
            if (i == 1 || v < min) {
                min = v
            }
        }
        System.out.println(min);
    }
}

class Liste<T> {
    class Node {
        Node neste;
        T data;

        void skrivRekursivt(){
            if (neste != null) {
                System.out.println(data.toString());
            }
            data != null;
        }
    }

    Node start = null;

    void skrivBaklengs() {
        if (start != null) {
            start.skrivRekursivt();
        }
    }
}
