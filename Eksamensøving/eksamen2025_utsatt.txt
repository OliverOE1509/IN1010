        første.neste = siste;
    }

    public void settInnBak(Bil erILista, Bil ny) {
        ny.neste = erILista.neste;
        erILista.neste = ny; 
    }

We need to write "ny.neste = erILista.neste" first, and then write "erILista.neste = ny", becuase "erILista.neste = ny" loses access to erILista.neste, which we want ny.neste to point at.

    if (ix < 0) {
        return a[0];
    }
    else if (ix >= a.length) {
        return a[a.length-1];

abstract class EnVeis {
    EnVeis neste;
}

class EnVeisBeholder<E extends EnVeis> {
    private E første, siste;

    public void settInnSist(E ny) {
        if (første == null) {
            første = ny; 
            siste = ny;
        } else {
            siste.neste = ny;
            ny.siste = ny;
        }
    }

    public E taUtFørst() {
        if (første == null) {
            return null;
        }
        E ut = første;
        if (første == siste) {
            første = null;
            siste = null;
        } else {
            første = (E) første.neste;
        }
        return ut;
    }
}

class Lager implements Iterable<String> {
    String[] innhold = new String[10];

    @Override
    public Iterator<String> iterator() {
        return new LagerIterator();
    }
}

class LagerIterator implements Iterator<String> {
    int ix = innhold.length - 1;

    @Override
    public boolean hasNext() {
        return ix >= 0;
    }

    @Override String next() {
        String res = innhold[ix];
        ix = ix - 1;
        return res;
    }
}
