import java.util.Scanner;

class SkrivTall {
    public static void main (String[] args) {
        int[] tallene = new int[10];
        int antallTall = 0;
        Scanner tastatur = new Scanner(System.in)
        while (antallTall < 10) {
            int v = tastatur.nextInt();
            if (v < 0) {
                break;
            }
            tallene[antallTall] = v;
            antallTall = antallTall + 1;

        }

        for (int i = antallTall - 1; i >= 0; i--) {
            System.out.println(tallene[i]);
        }
    }
}

class Dobbeltliste<T> {
    class Node {
        Node forrige, neste;
        T data;
    }

    Node første, siste;

    T fjernførste() {
        Node resP = første
        første = første.neste;
        return resP.data;
    }

    T fjernSiste() {
        Node resP = siste;
        siste = siste.forrige;
        return resP.data;
    }

    T hentNr(int n) {
        Node nodeP = første;
        for (int i = 0; i < n; i++) {
            nodeP = nodeP.neste;
        }
        return nodeP.data;
    }
}

class Bil {
    Bil forrige, neste;
}

class BilKø{
    Bil første, siste;

    void taUtInni(Bil b) {
        b.forrige.neste = b.neste; b.neste.forrige = b.forrige;
        b.neste = null; b.forrige = null;
    }

    void settInn(Bil b) {
        if (første == null) {
            første = b;
        } else {
            b.forrige = siste; siste.neste = b;
        }
        b = siste
    }
}
