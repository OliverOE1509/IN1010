        Hund h1, h2, h3;
        h1 = new Hund("Voff-voff");

        h2 = h1;
        h1 = null;
        h3 = h2;

class Bil {
    Bil forrige;
    Bil neste;
}

class BilListe {
    Bil første, siste;

    public void settInnFør(Bil ny, Bil erILista) {
        if (første == null) {
            første = ny; siste = ny;
        } else {
            ny.neste = erILista;
            if (første == erILista) {
                første.forrige = ny;
                første = ny;
            } else {
                ny.forrige = erILista.forrige;
                erILista.forrige.neste = ny;
                erILista.forrige = ny;
            }
        }
    }
}
